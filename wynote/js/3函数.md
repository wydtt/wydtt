变量屏蔽
```javascript
<script>
  var i=10;
  function fn1() {
  //屏蔽层
    var i=100;//这个定义阻挡了这个局部的变成了10000,但是alert的是全局的i所以是10
    function fn2() {
      i=1000;
      function fn3() {
        i=10000;
      }
      fn3();
    }
    fn2();
  }
  fn1();
  alert(i);
</script>
```
输出10
